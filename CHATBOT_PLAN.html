Project Overview

The Abona AI Chatbot is a Python-based microservice that adds intelligent automated responses to the existing Abona Shop live chat system. It operates alongside the Node.js backend without replacing or modifying the core architecture — it acts purely as an intelligent responder that Node.js consults when needed.

The bot handles two scenarios: product recommendations triggered by @abona mentions, and automatic courtesy replies when no admin responds within 30 seconds. All responses are delivered in real time through the existing Socket.io infrastructure.

System Architecture

The system follows a microservices pattern. The Node.js backend remains the single point of contact for the frontend. When bot logic is needed, Node.js makes an internal HTTP call to the Python service, receives a response, and delivers it to the customer via Socket.io.

Code Overview

Python — main.py (FastAPI endpoint)

from fastapi import FastAPI
from analyzer import detect_intent
from db import search_products
from responder import build_reply

app = FastAPI()

@app.post("/analyze")
async def analyze(req: dict):
    message = req.get("message", "")
    intent  = detect_intent(message)
    products = search_products(intent)
    reply   = build_reply(intent, products)
    return { "reply": reply, "products": products }

Python — analyzer.py (Intent Detection)

import spacy
nlp = spacy.load("en_core_web_sm")

INTENTS = {
    "greeting":  ["hi","hello","hey","สวัสดี"],
    "gift":      ["gift","present","surprise"],
    "budget":    ["cheap","budget","under","affordable"],
    "shipping":  ["shipping","delivery","arrive"],
    "returns":   ["return","refund","exchange"],
    "best":      ["popular","best","top","trending"],
}

def detect_intent(message: str) -> dict:
    doc  = nlp(message.lower())
    tokens = [t.text for t in doc]
    intent = "unknown"
    budget = None

    for name, keywords in INTENTS.items():
        if any(k in tokens for k in keywords):
            intent = name
            break

    # Extract budget amount (e.g. "under 1500")
    for i, token in enumerate(doc):
        if token.like_num and i > 0:
            budget = float(token.text.replace(",",""))

    return { "intent": intent, "budget": budget, "tokens": tokens }

Node.js — Auto-reply Timer (chat.js)

const timers = new Map();

socket.on('customer:message', async ({ conversationId, message }) => {
  // Start 30-second timer
  const timer = setTimeout(async () => {
    const res = await fetch('http://localhost:8000/analyze', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ message, conversation_id: conversationId })
    });
    const { reply } = await res.json();
    // Save bot message and emit to room
    saveBotMessage(conversationId, reply);
    io.to(`conv:${conversationId}`).emit('chat:message', { ... });
  }, 30000);

  timers.set(conversationId, timer);
});

// Cancel timer when admin replies
socket.on('admin:message', ({ conversationId }) => {
  clearTimeout(timers.get(conversationId));
  timers.delete(conversationId);
});
